// رابط السيرفر const API = "http://localhost:3000" ////////////////////////////////////////////////// // جلب التذكيرات من السيرفر ////////////////////////////////////////////////// export async function getReminders(token){ const res = await fetch(`${API}/api/reminders`,{ headers:{ Authorization:`Bearer ${token}` } }) return res.json() } ////////////////////////////////////////////////// // إنشاء تذكير جديد ////////////////////////////////////////////////// export async function createReminder(data,token){ const res = await fetch(`${API}/api/reminders`,{ method:"POST", headers:{ "Content-Type":"application/json", Authorization:`Bearer ${token}` }, body:JSON.stringify(data) }) return res.json() }
