// src/ai/dateResolver.js export function resolveDate(text) { const now = new Date() const lower = text.toLowerCase() let resultDate = new Date(now) if (lower.includes("اليوم")) { resultDate = new Date(now) } else if (lower.includes("بكرا") || lower.includes("غدا")) { resultDate.setDate(now.getDate() + 1) } else if (lower.includes("بعد بكرا")) { resultDate.setDate(now.getDate() + 2) } else if (lower.includes("بعد ساعة")) { resultDate.setHours(now.getHours() + 1) } else if (lower.includes("بعد يوم")) { resultDate.setDate(now.getDate() + 1) } else if (lower.includes("بعد يومين")) { resultDate.setDate(now.getDate() + 2) } else if (lower.includes("الأسبوع الجاي")) { resultDate.setDate(now.getDate() + 7) } else if (lower.includes("الشهر الجاي")) { resultDate.setMonth(now.getMonth() + 1) } else if (lower.includes("الجمعة")) { const today = now.getDay() const friday = 5 let diff = friday - today if (diff <= 0) diff += 7 resultDate.setDate(now.getDate() + diff) } return resultDate } export function resolveTime(text, date) { const match = text.match(/(\d{1,2})/) if (!match) return date let hour = parseInt(match[1]) if (text.includes("مساء")) { if (hour < 12) hour += 12 } if (text.includes("صباح")) { if (hour === 12) hour = 0 } date.setHours(hour) date.setMinutes(0) date.setSeconds(0) return date }
