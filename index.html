import express from "express"; import { parseMessage } from "../ai/intentParser.js"; import Reminder from "../models/reminderModel.js"; import { protect } from "../middlewares/authMiddleware.js"; const router = express.Router(); // AI message route router.post("/message", protect, async (req, res) => { try { const { message } = req.body; // تحليل الرسالة const parsed = parseMessage(message); // إذا كان الطلب إنشاء تذكير if (parsed.intent === "reminder") { const reminder = await Reminder.create({ user: req.user._id, text: parsed.text, time: parsed.time }); return res.json({ reply: تمام، ذكرتك: ${parsed.text}, reminder }); } // إذا لم يفهم الذكاء الاصطناعي return res.json({ reply: "لم أفهم الطلب." }); } catch (error) { console.error(error); res.status(500).json({ message: "Server error" }); } }); export default router;
