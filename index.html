const API_URL = "http://localhost:3000/api/reminders"; export const getReminders = async (token) => { try { const res = await fetch(API_URL, { method: "GET", headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}`, }, }); const data = await res.json(); return data; } catch (err) { console.error("getReminders error:", err); throw err; } }; export const createReminder = async (reminder, token) => { try { const res = await fetch(API_URL, { method: "POST", headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}`, }, body: JSON.stringify(reminder), }); const data = await res.json(); return data; } catch (err) { console.error("createReminder error:", err); throw err; } }; export const updateReminder = async (id, reminder, token) => { try { const res = await fetch(`${API_URL}/${id}`, { method: "PUT", headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}`, }, body: JSON.stringify(reminder), }); const data = await res.json(); return data; } catch (err) { console.error("updateReminder error:", err); throw err; } }; export const deleteReminder = async (id, token) => { try { const res = await fetch(`${API_URL}/${id}`, { method: "DELETE", headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}`, }, }); const data = await res.json(); return data; } catch (err) { console.error("deleteReminder error:", err); throw err; } };
