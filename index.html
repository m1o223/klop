import { detectIntent } from "./intentParser.js" import { extractTime } from "./timeExtractor.js" import { translateText } from "./translator.js" import { askAI } from "./openaiParser.js" import { detectLanguage } from "./languageDetector.js" export async function processAI(message) { const language = detectLanguage(message) const intent = detectIntent(message) const time = extractTime(message) // reminder intent if (intent === "reminder" && time) { return { intent: "reminder", time, reply: language === "ar" ? "تمام 👌 رح أذكرك بهذا الموعد" : "Got it 👌 I will remind you" } } // translate intent if (intent === "translate") { const translation = await translateText(message) return { intent: "translate", reply: translation } } // normal AI question const aiReply = await askAI(message) return { intent: "chat", reply: aiReply } }
