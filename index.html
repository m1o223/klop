import admin from "firebase-admin"; import { env } from "./env.js"; import fs from "fs"; function initFirebase() { if (admin.apps.length) return; // خيار 1: serviceAccount.json path if (env.FIREBASE_SERVICE_ACCOUNT_PATH) { const path = env.FIREBASE_SERVICE_ACCOUNT_PATH; if (!fs.existsSync(path)) { throw new Error( `Firebase service account file not found: ${path} (check FIREBASE_SERVICE_ACCOUNT_PATH)` ); } const raw = fs.readFileSync(path, "utf-8"); const serviceAccount = JSON.parse(raw); admin.initializeApp({ credential: admin.credential.cert(serviceAccount), }); return; } // خيار 2: Env vars (للنشر) if (env.FIREBASE_PROJECT_ID && env.FIREBASE_CLIENT_EMAIL && env.FIREBASE_PRIVATE_KEY) { admin.initializeApp({ credential: admin.credential.cert({ projectId: env.FIREBASE_PROJECT_ID, clientEmail: env.FIREBASE_CLIENT_EMAIL, privateKey: env.FIREBASE_PRIVATE_KEY.replace(/\\n/g, "\n"), }), }); return; } throw new Error( "Firebase Admin not configured. Provide FIREBASE_SERVICE_ACCOUNT_PATH or FIREBASE_PROJECT_ID/CLIENT_EMAIL/PRIVATE_KEY" ); } initFirebase(); export { admin }; export const db = admin.firestore(); export const auth = admin.auth();
