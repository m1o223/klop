const API = "http://localhost:3000/api/reminders"; export const getReminders = async (token) => { if (!token) { throw new Error("No auth token"); } const res = await fetch(API, { method: "GET", headers: { Authorization: `Bearer ${token}` } }); const data = await res.json(); if (!res.ok) { throw new Error(data.message || "Failed to fetch reminders"); } return data; }; export const createReminder = async (data, token) => { if (!token) { throw new Error("No auth token"); } const res = await fetch(API, { method: "POST", headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}` }, body: JSON.stringify(data) }); const result = await res.json(); if (!res.ok) { throw new Error(result.message || "Failed to create reminder"); } return result; };
