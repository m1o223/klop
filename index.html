import * as chrono from "chrono-node"; /** * BlueMind AI - Ultra Intent Parser (AR + EN) * ------------------------------------------------ * ميزات: * - فهم وقت طبيعي عربي/إنجليزي (chrono-node) * - دعم أجزاء الثانية (ISO + timestamps + hh:mm:ss) * - دعم تكرار: يومي/أسبوعي/شهري/سنوي + كل X دقائق/ساعات/أيام/أسابيع * - فصل المهمة عن الوقت (task cleaning) * - timezone support (افتراضي UTC أو ما تمرره أنت) * - forwardDate true لتجنب ماضي غير مقصود * - fallback ذكي إذا ما كتب وقت * * مخرجات: * { * intent: "reminder" | "unknown", * task: string, * timeISO: string, // ISO string with ms * time: Date, * repeat: null | { type, every, unit, at?, byweekday? }, * meta: { timezone, confidence } * } */ const REMINDER_TRIGGERS = [ "ذكرني", "ذكّرني", "remind me", "remind", ]; const AR_WEEKDAYS = { "الاثنين": "MO", "الإثنين": "MO", "الثلاثاء": "TU", "الأربعاء": "WE", "الخميس": "TH", "الجمعة": "FR", "السبت": "SA", "الأحد": "SU", }; const EN_WEEKDAYS = { "monday": "MO", "tuesday": "TU", "wednesday": "WE", "thursday": "TH", "friday": "FR", "saturday": "SA", "sunday": "SU", }; function includesAny(haystack, arr) { const t = haystack.toLowerCase(); return arr.some(w => t.includes(w)); } /** * تحويل "بعد 10 دقائق/ساعتين/3 أيام/أسبوع" إلى milliseconds */ function parseRelativeEvery(text) { // عربي: كل/بعد + رقم + وحدة // إنجليزي: every/in + number + unit const t = text.toLowerCase(); // كل ساعتين / كل 2 ساعة / كل 10 دقائق / كل يوم / كل أسبوع let m = t.match(/(?:كل)\s+(\d+)\s*(ثانية|ثواني|دقيقة|دقائق|ساعة|ساعات|يوم|أيام|أسبوع|أسابيع|شهر|شهور|سنة|سنوات)/i); if (m) { const every = parseInt(m[1], 10); const unit = normalizeUnit(m[2]); return { every, unit }; } // كل يوم / كل أسبوع / كل شهر / كل سنة m = t.match(/(?:كل)\s+(يوم|أسبوع|شهر|سنة)/i); if (m) { const unit = normalizeUnit(m[1]); return { every: 1, unit }; } // every 10 minutes / every 2 hours m = t.match(/every\s+(\d+)\s*(second|seconds|minute|minutes|hour|hours|day|days|week|weeks|month|months|year|years)/i); if (m) { const every = parseInt(m[1], 10); const unit = normalizeUnit(m[2]); return { every, unit }; } // every day / every week / every month / every year m = t.match(/every\s+(day|week|month|year)/i); if (m) { const unit = normalizeUnit(m[1]); return { every: 1, unit }; } return null; } function normalizeUnit(u) { const x = (u || "").toLowerCase(); // عربي if (["ثانية","ثواني"].includes(x)) return "second"; if (["دقيقة","دقائق"].includes(x)) return "minute"; if (["ساعة","ساعات"].includes(x)) return "hour"; if (["يوم","أيام"].includes(x)) return "day"; if (["أسبوع","اسبوع","أسابيع","اسابيع"].includes(x)) return "week"; if (["شهر","شهور"].includes(x)) return "month"; if (["سنة","سنوات"].includes(x)) return "year"; // إنجليزي if (x.startsWith("second")) return "second"; if (x.startsWith("minute")) return "minute"; if (x.startsWith("hour")) return "hour"; if (x.startsWith("day")) return "day"; if (x.startsWith("week")) return "week"; if (x.startsWith("month")) return "month"; if (x.startsWith("year")) return "year"; return "minute"; } /** * استخراج وقت HH:MM(:SS(.ms)) إذا مكتوب صراحة */ function extractExplicitClock(text) { // 09:30 / 9:30 / 09:30:15 / 09:30:15.250 const m = text.match(/\b(\d{1,2}):(\d{2})(?::(\d{2})(?:\.(\d{1,3}))?)?\b/); if (!m) return null; const hh = parseInt(m[1], 10); const mm = parseInt(m[2], 10); const ss = m[3] ? parseInt(m[3], 10) : 0; const ms = m[4] ? parseInt(m[4].padEnd(3, "0"), 10) : 0; if (hh > 23 || mm > 59 || ss > 59) return null; return { hh, mm, ss, ms, raw: m[0] }; } /** * استخراج يوم أسبوع (AR/EN) */ function extractWeekday(text) { const t = text.toLowerCase(); for (const [k, v] of Object.entries(AR_WEEKDAYS)) { if (t.includes(k.toLowerCase())) return v; } for (const [k, v] of Object.entries(EN_WEEKDAYS)) { if (t.includes(k)) return v; } return null; } /** * تنظيف المهمة بإزالة كلمات التفعيل + كلمات وقت عامة + أجزاء chrono match */ function cleanTask(original, chronoSpanText = "") { let task = original; // إزالة trigger REMINDER_TRIGGERS.forEach(w => { const re = new RegExp(w.replace(/[.*+?^${}()|[\]\\]/g, "\\$&"), "ig"); task = task.replace(re, " "); }); // إزالة بعض كلمات الوقت العامة const generalTimeTokens = [ "بعد","بكرة","غداً","غدا","اليوم","الساعة","ساعه","عند","في", "tomorrow","today","at","on","in","next","this", "minute","minutes","hour","hours","day","days","week","weeks","month","months","year","years", "كل","every" ]; generalTimeTokens.forEach(tk => { const re = new RegExp(`\\b${tk}\\b`, "ig"); task = task.replace(re, " "); }); // إزالة النص الذي التقطه chrono إن وجد if (chronoSpanText) { const escaped = chronoSpanText.replace(/[.*+?^${}()|[\]\\]/g, "\\$&"); task = task.replace(new RegExp(escaped, "ig"), " "); } // تنظيف مسافات task = task.replace(/\s+/g, " ").trim(); return task; } /** * تطبيق HH:MM(:SS.ms) على تاريخ معيّن */ function applyClock(date, clock) { const d = new Date(date); d.setHours(clock.hh, clock.mm, clock.ss, clock.ms); return d; } /** * ضمان forward (لو طلع وقت بالماضي بالغلط) */ function ensureFuture(date, now) { if (!date) return null; const d = new Date(date); if (d.getTime() < now.getTime()) { // ادفعه للأمام ساعة كحد أدنى بدل ما يكون ماضي return new Date(now.getTime() + 60 * 60 * 1000); } return d; } export function parseMessage(message, opts = {}) { const now = opts.now ? new Date(opts.now) : new Date(); const timezone = opts.timezone || "UTC"; // لاحقاً ممكن تربطه بتايم زون المستخدم const raw = String(message || "").trim(); const lower = raw.toLowerCase(); // هل هو تذكير؟ if (!includesAny(lower, REMINDER_TRIGGERS)) { return { intent: "unknown" }; } // 1) chrono parse const results = chrono.parse(raw, now, { forwardDate: true }); const chronoHit = results?.[0] || null; let time = chronoHit ? chronoHit.start.date() : null; let chronoText = chronoHit ? (chronoHit.text || "") : ""; // 2) explicit clock overrides (يدعم ثواني/ميلي ثانية) const clock = extractExplicitClock(raw); if (clock) { // إذا chrono ما أعطى يوم، استخدم اليوم const base = time || now; time = applyClock(base, clock); chronoText = chronoText || clock.raw; } // 3) إذا ذكر يوم أسبوع بدون تاريخ واضح، chrono غالباً يفهمه، بس نأكده const byweekday = extractWeekday(raw); // 4) repeat parsing const every = parseRelativeEvery(raw); let repeat = null; if (every) { repeat = { type: "interval", every: every.every, unit: every.unit }; } else if (byweekday && (raw.includes("كل") || raw.toLowerCase().includes("every"))) { repeat = { type: "weekly", byweekday }; } // 5) fallback time لو ما في if (!time) { // افتراض: بعد ساعة time = new Date(now.getTime() + 60 * 60 * 1000); } time = ensureFuture(time, now); // 6) task extraction const task = cleanTask(raw, chronoText) || raw; // 7) confidence (بسيطة) const confidence = chronoHit || clock ? 0.85 : 0.6; return { intent: "reminder", task, time, timeISO: time.toISOString(), // يحتوي milliseconds repeat, meta: { timezone, confidence } }; }
