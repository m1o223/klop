// ============================= // server.js // ============================= import express from "express" import dotenv from "dotenv" import cors from "cors" import mongoose from "mongoose" // ============================= // Routes // ============================= import authRoutes from "./src/routes/authRoutes.js" import userRoutes from "./src/routes/userRoutes.js" import chatRoutes from "./src/routes/chatRoutes.js" import reminderRoutes from "./src/routes/reminderRoutes.js" import stripeRoutes from "./src/routes/stripeRoutes.js" import zakatRoutes from "./src/routes/zakatRoutes.js" import adminRoutes from "./src/routes/adminRoutes.js" import aiRoutes from "./src/routes/aiRoutes.js" // ============================= // Config // ============================= dotenv.config() const app = express() // ============================= // Middlewares // ============================= app.use(cors()) app.use(express.json()) // ============================= // Health Check // ============================= app.get("/", (req, res) => { res.json({ message: "BlueMind API is running 🚀" }) }) // ============================= // API Routes // ============================= app.use("/api/auth", authRoutes) app.use("/api/users", userRoutes) app.use("/api/chat", chatRoutes) app.use("/api/reminders", reminderRoutes) app.use("/api/stripe", stripeRoutes) app.use("/api/zakat", zakatRoutes) app.use("/api/admin", adminRoutes) app.use("/api/ai", aiRoutes) // ============================= // Not Found // ============================= app.use((req, res) => { res.status(404).json({ message: `Route not found: ${req.originalUrl}` }) }) // ============================= // Error Handler // ============================= app.use((err, req, res, next) => { console.error(err) const statusCode = res.statusCode && res.statusCode !== 200 ? res.statusCode : 500 res.status(statusCode).json({ message: err.message || "Server error" }) }) // ============================= // Server + MongoDB // ============================= const PORT = process.env.PORT || 3000 const startServer = async () => { console.log("Starting server...") try { if (!process.env.MONGO_URI) { console.log("❌ MONGO_URI missing in .env") process.exit(1) } console.log("Connecting to MongoDB...") await mongoose.connect(process.env.MONGO_URI) console.log("✅ MongoDB Connected") app.listen(PORT, () => { console.log(`🚀 Server running on port ${PORT}`) }) } catch (error) { console.log("❌ MongoDB connection error:", error.message) process.exit(1) } } startServer()
