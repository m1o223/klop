const API = "http://localhost:3000/api/reminders" function getToken() { return localStorage.getItem("token") } export async function getReminders() { const res = await fetch(API, { headers: { Authorization: `Bearer ${getToken()}` } }) return res.json() } export async function createReminder(reminder) { await fetch(API, { method: "POST", headers: { "Content-Type": "application/json", Authorization: `Bearer ${getToken()}` }, body: JSON.stringify(reminder) }) } export async function deleteReminder(id) { await fetch(`${API}/${id}`, { method: "DELETE", headers: { Authorization: `Bearer ${getToken()}` } }) }
