import express from "express"; import { ensureUser } from "../services/userService.js"; import { verifyFirebaseToken } from "../middlewares/auth.js"; const router = express.Router(); router.post("/create-user", verifyFirebaseToken, async (req, res) => { try { const uid = req.body.uid || req.auth?.uid; const email = req.body.email || req.auth?.email || null; if (!uid) return res.status(400).json({ error: "uid required" }); const user = await ensureUser(uid, email); res.json({ success: true, user }); } catch (err) { res.status(500).json({ error: err.message }); } }); export default router;
