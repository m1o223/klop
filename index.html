import { useState, useEffect } from "react"; import { getReminders, createReminder, updateReminder, deleteReminder, } from "../api/remindersApi"; export default function Reminders() { const token = localStorage.getItem("token"); const [reminders, setReminders] = useState([]); const [showModal, setShowModal] = useState(false); const [menuOpen, setMenuOpen] = useState(null); const [editingId, setEditingId] = useState(null); const [title, setTitle] = useState(""); const [description, setDescription] = useState(""); const [date, setDate] = useState(""); const [hour, setHour] = useState(""); const [minute, setMinute] = useState(""); const [before, setBefore] = useState("5"); const resetForm = () => { setTitle(""); setDescription(""); setDate(""); setHour(""); setMinute(""); setBefore("5"); setEditingId(null); }; const closeModal = () => { setShowModal(false); resetForm(); }; // ⭐ إصلاح مشكلة الوقت const formatTime = (time) => { if (!time) return ""; const h = String(time.hour ?? 0).padStart(2, "0"); const m = String(time.minute ?? 0).padStart(2, "0"); return `${h}:${m}`; }; const loadReminders = async () => { try { const data = await getReminders(token); console.log("API DATA:", data); setReminders(data.items || []); } catch (err) { console.error("load reminders error:", err); } }; useEffect(() => { if (token) { loadReminders(); } }, [token]); const handleCreateOrUpdate = async (e) => { e.preventDefault(); try { const payload = { title, description, date, time: { hour: Number(hour), minute: Number(minute), }, }; if (editingId) { await updateReminder(editingId, payload, token); } else { await createReminder(payload, token); } await loadReminders(); closeModal(); } catch (err) { console.error(err); } };
