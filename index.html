const API = "http://localhost:3000/api/reminders" export const getReminders = async (token) => { const res = await fetch(API, { headers: { Authorization: `Bearer ${token}` } }) return res.json() } export const createReminder = async (data, token) => { const res = await fetch(API, { method: "POST", headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}` }, body: JSON.stringify(data) }) return res.json() }
