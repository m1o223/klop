import parseMessage from "../ai/intentParser.js"; import Reminder from "../models/reminderModel.js"; export const chat = async (req, res) => { try { const { message } = req.body; // تحليل الرسالة const result = parseMessage(message); if (result.intent === "reminder") { const reminder = await Reminder.create({ user: req.user._id, text: result.text, time: result.time }); return res.json({ reply: `تمام، سأذكرك: ${result.text}` }); } return res.json({ reply: "لم أفهم الطلب." }); } catch (error) { res.status(500).json({ error: error.message }); } };
