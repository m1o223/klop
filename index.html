import { parseIntent } from "../ai/intentParser.js" import { resolveDate, resolveTime } from "../ai/dateResolver.js" import { askAI } from "../ai/openaiParser.js" import Reminder from "../models/reminderModel.js" export const chatWithAI = async (req, res) => { try { const { message, userId } = req.body const intent = parseIntent(message) // إذا كان تذكير if (intent.intent === "reminder") { let date = resolveDate(message) date = resolveTime(message, date) const reminder = await Reminder.create({ user: userId, text: message, date }) return res.json({ reply: `تمام 👍 ذكرتك: ${message}`, reminder }) } // إذا طلب عرض التذكيرات if (intent.intent === "list_reminders") { const reminders = await Reminder.find({ user: userId }).sort({ date: 1 }) return res.json({ reminders }) } // غير ذلك → نرسل الرسالة للذكاء const aiReply = await askAI(message) return res.json({ reply: aiReply }) } catch (error) { console.error(error) res.status(500).json({ message: "AI Error" }) } }
