import { useEffect, useState } from "react"; import { useNavigate } from "react-router-dom"; import { auth } from "../config/firebase"; export default function Reminders() { const navigate = useNavigate(); const API = "http://localhost:3000/api/reminders"; const [reminders, setReminders] = useState([]); const [title, setTitle] = useState(""); const [description, setDescription] = useState(""); const [showModal, setShowModal] = useState(false); const [editId, setEditId] = useState(null); /* ============================= GET FIREBASE TOKEN ============================= */ const getToken = async () => { return new Promise((resolve, reject) => { const unsubscribe = auth.onAuthStateChanged(async (user) => { if (!user) { reject("No user logged in"); return; } try { const token = await user.getIdToken(true); resolve(token); unsubscribe(); } catch (err) { reject(err); } }); }); }; /* ============================= FETCH REMINDERS ============================= */ const fetchReminders = async () => { try { const token = await getToken(); const res = await fetch(API, { headers: { Authorization: `Bearer ${token}` } }); const data = await res.json(); setReminders(Array.isArray(data) ? data : data.items || []); } catch (err) { console.error(err); } }; /* ============================= CREATE REMINDER ============================= */ const createReminder = async () => { try { const token = await getToken(); await fetch(API, { method: "POST", headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}` }, body: JSON.stringify({ title, description }) }); setTitle(""); setDescription(""); setShowModal(false); fetchReminders(); } catch (err) { console.error(err); } }; /* ============================= DELETE REMINDER ============================= */ const deleteReminder = async (id) => { try { const token = await getToken(); await fetch(`${API}/${id}`, { method: "DELETE", headers: { Authorization: `Bearer ${token}` } }); fetchReminders(); } catch (err) { console.error(err); } }; /* ============================= UPDATE REMINDER ============================= */ const updateReminder = async () => { try { const token = await getToken(); await fetch(`${API}/${editId}`, { method: "PUT", headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}` }, body: JSON.stringify({ title, description }) }); setEditId(null); setTitle(""); setDescription(""); setShowModal(false); fetchReminders(); } catch (err) { console.error(err); } }; /* ============================= OPEN EDIT MODAL ============================= */ const openEdit = (reminder) => { setEditId(reminder._id); setTitle(reminder.title); setDescription(reminder.description); setShowModal(true); }; /* ============================= PAGE LOAD ============================= */ useEffect(() => { fetchReminders(); }, []);
